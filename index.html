Sub SmartProductTitleRemap() Dim ws As Worksheet Dim lastRow As Long Dim i As Long Dim title As String Dim sku As String Dim productSuffix As String Dim qtyString As String Dim maxTitleLen As Long Set ws = ActiveSheet lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row ' Speed optimize Excel settings for 55,000 records Application.ScreenUpdating = False Application.Calculation = xlCalculationManual For i = 2 To lastRow title = Trim(CStr(ws.Cells(i, 1).Value)) sku = LCase(Trim(CStr(ws.Cells(i, 4).Value))) ' Read Column 4 (Column D) If sku <> "" Then ' ---------------------------------------------------- ' PHASE 1: CLEAN COLUMN A (Remove existing product terms) ' ---------------------------------------------------- title = Replace(title, " - rect keychain", "") title = Replace(title, " - Sq Keychain", "") title = Replace(title, "keychain", "") title = Replace(title, "key chain", "") title = Replace(title, "coaster", "") title = Replace(title, "mousepad", "") title = Replace(title, "ornament", "") title = Replace(title, "cutting board", "") title = Replace(title, "dog tag", "") title = Replace(title, "card holder", "") title = Replace(title, "switch plate", "") title = Replace(title, "ceramic tile", "") title = Replace(title, "tile", "") title = Replace(title, " x12", "") title = Replace(title, " x8", "") title = Replace(title, " x4", "") title = Replace(title, " x2", "") title = Replace(title, " (x12)", "") title = Replace(title, " (x8)", "") title = Replace(title, " (x4)", "") title = Replace(title, " (x2)", "") title = Replace(title, " (4)", "") title = Replace(title, " (8)", "") ' Strip out extra leftover dashes or trailing spaces from scrubbing title = Trim(title) If Left(title, 1) = "-" Then title = Trim(Mid(title, 2)) If Right(title, 1) = "-" Then title = Trim(Left(title, Len(title) - 1)) title = Trim(title) ' ---------------------------------------------------- ' PHASE 2: DETERMINE QUANTITY (Skip if SKU contains 'mas') ' ---------------------------------------------------- qtyString = "" If InStr(sku, "mas") = 0 Then If InStr(sku, "x12") > 0 Or InStr(sku, "12x") > 0 Then qtyString = " (x12)" _ Else If InStr(sku, "x8") > 0 Or InStr(sku, "8x") > 0 Then qtyString = " (x8)" _ Else If InStr(sku, "x4") > 0 Or InStr(sku, "4x") > 0 Then qtyString = " (x4)" _ Else If InStr(sku, "x2") > 0 Or InStr(sku, "2x") > 0 Then qtyString = " (x2)" End If ' ---------------------------------------------------- ' PHASE 3: INTERPRET PRODUCT TRANSLATION RULES (Syntax Corrected) ' ---------------------------------------------------- productSuffix = "" ' Coaster Rules If InStr(sku, "sftcoasterround") > 0 Then productSuffix = "RND coaster" & qtyString ElseIf InStr(sku, "sftcoastersq") > 0 Then productSuffix = "Sq Coasters" & qtyString ElseIf InStr(sku, "coaster") > 0 And InStr(sku, "mas") > 0 Then productSuffix = "coasters" ' Tile Rules ElseIf InStr(sku, "tile") > 0 Then If InStr(sku, "4in") > 0 Or InStr(sku, "4-inch") > 0 Then productSuffix = "4in Ceramic Tile" _ Else If InStr(sku, "6in") > 0 Or InStr(sku, "6-inch") > 0 Then productSuffix = "6in Ceramic Tile" _ Else If InStr(sku, "8in") > 0 Or InStr(sku, "8-inch") > 0 Then productSuffix = "8in Ceramic Tile" _ Else productSuffix = "Ceramic Tile" ' Keychain Rules ElseIf InStr(sku, "keychain") > 0 Or InStr(sku, "key chain") > 0 Then If InStr(sku, "rect") > 0 Then productSuffix = "Rectangle keychain" & qtyString _ Else If InStr(sku, "sq") > 0 Then productSuffix = "Square keychain" & qtyString _ Else productSuffix = "Keychain" & qtyString ' Switch Plate Rules ElseIf InStr(sku, "switch") > 0 Or InStr(sku, "plate") > 0 Then If InStr(sku, "single toggle") > 0 Or InStr(sku, "single-toggle") > 0 Then productSuffix = "single toggle" _ Else If InStr(sku, "double toggle") > 0 Or InStr(sku, "double-toggle") > 0 Then productSuffix = "double toggle" _ Else If InStr(sku, "gfi") > 0 Then productSuffix = "gfi" _ Else If InStr(sku, "dbl rocker") > 0 Then productSuffix = "dbl rocker" _ Else If InStr(sku, "rocker") > 0 Then productSuffix = "rocker" _ Else productSuffix = "Switch Plate" ' Miscellaneous Products Rules ElseIf InStr(sku, "ornament") > 0 Then productSuffix = "Ornament" & qtyString ElseIf InStr(sku, "cutting") > 0 Or InStr(sku, "board") > 0 Then productSuffix = "Cutting Board" & qtyString ElseIf InStr(sku, "dog") > 0 Or InStr(sku, "tag") > 0 Then productSuffix = "Dog Tag" & qtyString ElseIf InStr(sku, "card") > 0 Or InStr(sku, "holder") > 0 Then productSuffix = "Card Holder" & qtyString End If ' ---------------------------------------------------- ' PHASE 4: REBUILD AND ENFORCE 70-CHAR LIMIT ' ---------------------------------------------------- If productSuffix <> "" Then maxTitleLen = 70 - Len(productSuffix) - 3 If Len(title) > maxTitleLen Then title = Trim(Left(title, maxTitleLen)) If Right(title, 1) = "," Or Right(title, 1) = "-" Then title = Trim(Left(title, Len(title) - 1)) End If End If ws.Cells(i, 1).Value = title & " - " & productSuffix End If End If Next i ' Re-enable application speeds Application.Calculation = xlCalculationAutomatic Application.ScreenUpdating = True MsgBox "All 55,000 rows have been reformatted successfully!", vbInformation End Sub
